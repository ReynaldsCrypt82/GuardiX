    ```ts
    export async function updateGroupAction(
      slug: string,
      groupId: string,
      formData: FormData,
    ) {
      const raw = Object.fromEntries(formData) as Record<string, unknown>
      // Normalizar next_assembly_date vazio → null (CON-05 — campo nullable)
      if (raw.next_assembly_date === '') raw.next_assembly_date = null

      const parsed = updateGroupSchema.safeParse(raw)
      if (!parsed.success) {
        const fieldErrors = parsed.error.flatten().fieldErrors
        const firstKey = Object.keys(fieldErrors)[0]
        const firstMsg = firstKey ? fieldErrors[firstKey]?.[0] : 'Dados inválidos.'
        return { error: firstMsg ?? 'Dados inválidos.' }
      }

      // eslint-disable-next-line @typescript-eslint/no-explicit-any
      const supabase = (await createClient()) as any
      const { data: { user } } = await supabase.auth.getUser()
      if (!user) return { error: 'Sessão expirada.' }

      const role = (user.app_metadata as { role?: string })?.role
      if (!['admin', 'corretor'].includes(role ?? '')) {
        return { error: 'Sem permissão para editar grupos.' }
      }

      const { error } = await supabase
        .from('consortium_groups')
        .update({
          administrator: parsed.data.administrator,
          type: parsed.data.type,
          credit_value: parsed.data.credit_value,
          term_months: parsed.data.term_months,
          total_quotas: parsed.data.total_quotas,
          next_assembly_date: parsed.data.next_assembly_date ?? null,
          updated_at: new Date().toISOString(),
        })
        .eq('id', groupId)
        .is('deleted_at', null) // Defensive — RLS already enforces tenant isolation

      if (error) return { error: 'Erro ao atualizar grupo.' }

      revalidatePath(`/${slug}/consorcio`)
      revalidatePath(`/${slug}/consorcio/${groupId}`)
      return { success: true }
    }
    ```

    Add the import at the top: `import { updateGroupSchema } from '@/lib/validations/consortium-schemas'`

    DO NOT touch `createGroupAction` — leave existing behavior unchanged.

    Per CON-05: nullable `next_assembly_date` is the critical contract — schema must accept `null`, action must persist `null`, RLS already enforces tenant isolation.

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Browser → Server Action | FormData crosses from untrusted client to server; tenant_id MUST come from auth JWT, never FormData |
| Server Action → Postgres (RLS) | Supabase RLS policies enforce tenant isolation on `consortium_groups` table |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-03-06-01 | Tampering | updateGroupAction FormData | mitigate | Zod safeParse on every field via `updateGroupSchema`; reject on validation failure with single string error (Task 1) |
| T-03-06-02 | Information Disclosure | Cross-tenant edit attempt (user A edits group of tenant B by passing different groupId) | mitigate | Existing RLS policy on `consortium_groups` filters by `tenant_id = auth.jwt() -> 'app_metadata' -> 'tenant_id'` — supabase update with `.eq('id', groupId)` cannot bypass RLS; supplemented by `.is('deleted_at', null)` to block edits to soft-deleted groups (Task 1) |
| T-03-06-03 | Elevation of Privilege | Visualizador role attempts to edit group | mitigate | Server-side role guard `['admin', 'corretor']` in updateGroupAction (preserved from existing implementation) |
| T-03-06-04 | Repudiation | Group edited but no audit trail of who/when | accept | `updated_at` timestamp persisted; full audit log deferred to Phase 7 per ROADMAP. Low value target (assembly date is operational data, not financial) |
| T-03-06-05 | Denial of Service | Spamming updateGroupAction | accept | Supabase free-tier rate limits suffice for solo-corretor scale (5k clients, 10k apolices); platform-level mitigation deferred |
| T-03-06-06 | Spoofing | FormData claims tenant_id of victim | mitigate | `updateGroupAction` does NOT read tenant_id from FormData — only uses `auth.getUser()` JWT app_metadata; RLS enforces actual tenant isolation at DB layer (preserved pattern from existing action) |
</threat_model>
