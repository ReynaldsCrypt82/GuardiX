       export function generateSlug(companyName: string, existingSlugs: string[] = []): string {
         const base = companyName
           .normalize('NFD')
           .replace(/[\u0300-\u036f]/g, '')
           .toLowerCase()
           .replace(/[^a-z0-9]+/g, '-')
           .replace(/^-+|-+$/g, '')
           .slice(0, 40) || 'corretora'
         if (!existingSlugs.includes(base)) return base
         let i = 2
         while (existingSlugs.includes(`${base}-${i}`)) i++
         return `${base}-${i}`
       }

       import { validateCNPJ, stripCNPJ } from '@/lib/validations/cnpj'

       export const revalidate = 3600

       export async function GET(
         _req: Request,
         ctx: { params: Promise<{ cnpj: string }> }
       ) {
         const { cnpj } = await ctx.params
         const digits = stripCNPJ(cnpj)
         if (!validateCNPJ(digits)) {
           return NextResponse.json({ error: 'CNPJ invĂˇlido' }, { status: 400 })
         }
         const controller = new AbortController()
         const timeout = setTimeout(() => controller.abort(), 3000)
         try {
           const res = await fetch(`https://brasilapi.com.br/api/cnpj/v1/${digits}`, {
             signal: controller.signal,
             next: { revalidate: 3600 },
           })
           clearTimeout(timeout)
           if (!res.ok) {
             return NextResponse.json(
               { error: 'CNPJ nĂŁo encontrado na Receita Federal. Preencha a razĂŁo social manualmente.' },
               { status: res.status }
             )
           }
           const data = await res.json()
           return NextResponse.json({
             razao_social: data.razao_social,
             nome_fantasia: data.nome_fantasia,
             situacao_cadastral: data.descricao_situacao_cadastral,

       'use server'
       import { redirect } from 'next/navigation'
       import { createAdminClient } from '@/lib/supabase/admin'
       import { createClient } from '@/lib/supabase/server'
       import { generateSlug } from '@/lib/utils/slug'
       import { registerFullSchema, loginSchema, type FormError } from '@/lib/validations/auth-schemas'

       export async function registerTenant(formData: FormData): Promise<{ error?: FormError } | void> {
         const raw = Object.fromEntries(formData)
         // Coerce acceptTerms ("on"/"true") to boolean
         raw.acceptTerms = raw.acceptTerms === 'on' || raw.acceptTerms === 'true' ? true : false as never
         const parsed = registerFullSchema.safeParse(raw)
         if (!parsed.success) return { error: parsed.error.flatten().fieldErrors as FormError }

         const { companyName, cnpj, segment, adminName, email, password } = parsed.data
         const admin = createAdminClient()

         // 1. Pre-check CNPJ uniqueness (friendlier error than FK violation)
         const { data: existing } = await admin.from('tenants').select('id').eq('cnpj', cnpj).is('deleted_at', null).maybeSingle()
         if (existing) return { error: { cnpj: ['Este CNPJ jĂˇ possui uma conta no NEXUS. FaĂ§a login ou recupere o acesso.'] } }

         // 2. Generate unique slug
         const { data: slugRows } = await admin.from('tenants').select('slug')
         const existingSlugs = (slugRows ?? []).map((r) => r.slug)
         const slug = generateSlug(companyName, existingSlugs)

         // 3. INSERT tenant
         const { data: tenant, error: tenantErr } = await admin
           .from('tenants')
           .insert({ name: companyName, slug, cnpj, segment })
           .select()
           .single()
         if (tenantErr || !tenant) return { error: { _form: ['Ocorreu um erro ao criar sua corretora. Tente novamente.'] } }

         // 4. Create auth user with app_metadata pre-populated
         const { data: authRes, error: authErr } = await admin.auth.admin.createUser({
           email,
           password,
           email_confirm: true,
           app_metadata: { tenant_id: tenant.id, role: 'admin', slug: tenant.slug },
           user_metadata: { full_name: adminName },
         })
         if (authErr || !authRes?.user) {
           // ROLLBACK â€” delete the tenant we just created (service_role bypasses prevent_hard_delete trigger)
           await admin.from('tenants').delete().eq('id', tenant.id)
           const isDupe = authErr?.message.toLowerCase().includes('already registered') || authErr?.message.toLowerCase().includes('duplicate')
           if (isDupe) return { error: { email: ['Este e-mail jĂˇ estĂˇ em uso. Tente fazer login ou use outro e-mail.'] } }
           return { error: { _form: ['Ocorreu um erro ao criar o usuĂˇrio admin. Tente novamente.'] } }
         }

         // 5. INSERT profile
         const { error: profileErr } = await admin.from('profiles').insert({
           id: authRes.user.id,
           tenant_id: tenant.id,
           full_name: adminName,
           role: 'admin',
         })
         if (profileErr) {
           // ROLLBACK cascade: auth user + tenant
           await admin.auth.admin.deleteUser(authRes.user.id)
           await admin.from('tenants').delete().eq('id', tenant.id)
           return { error: { _form: ['Ocorreu um erro ao finalizar o cadastro. Tente novamente.'] } }
         }

         // 6. Auto-login (on the ssr server client so cookies are set)
         const supabase = await createClient()
         const { error: signInErr } = await supabase.auth.signInWithPassword({ email, password })
         if (signInErr) {
           // User created but login failed â€” direct them to login manually
           redirect(`/login?next=/${slug}/dashboard`)
         }
         redirect(`/${slug}/dashboard`)
       }

       export async function loginWithPassword(formData: FormData): Promise<{ error?: FormError } | void> {
         const parsed = loginSchema.safeParse(Object.fromEntries(formData))
         if (!parsed.success) return { error: parsed.error.flatten().fieldErrors as FormError }

         const supabase = await createClient()
         const { error } = await supabase.auth.signInWithPassword(parsed.data)
         if (error) {
           return { error: { _form: ['E-mail ou senha incorretos. Verifique seus dados e tente novamente.'] } }
         }

         // Read claims to discover slug for redirect
         const anySupa = supabase.auth as unknown as { getClaims?: () => Promise<{ data: { claims: { app_metadata?: { slug?: string } } | null } | null }> }
         let slug: string | undefined
         if (typeof anySupa.getClaims === 'function') {
           const { data } = await anySupa.getClaims()
           slug = data?.claims?.app_metadata?.slug
         }
         if (!slug) {
           const { data } = await supabase.auth.getUser()
           slug = (data.user?.app_metadata as { slug?: string } | undefined)?.slug
         }
         const next = formData.get('next')?.toString()
         if (next && typeof next === 'string' && next.startsWith('/')) redirect(next)
         redirect(slug ? `/${slug}/dashboard` : '/login')
       }

       export async function signOut(): Promise<void> {
         const supabase = await createClient()
         await supabase.auth.signOut()
         redirect('/login')
       }

       export async function acceptInvite(token: string, formData: FormData): Promise<{ error?: string } | void> {
         const parsed = acceptInviteSchema.safeParse(Object.fromEntries(formData))
         if (!parsed.success) {
           const flat = parsed.error.flatten()
           const first = Object.values(flat.fieldErrors).flat()[0] ?? 'Dados invĂˇlidos'
           return { error: first }
         }
         const { fullName, password } = parsed.data
         const admin = createAdminClient()

         // ATOMIC single-use check â€” UPDATE returns the row only if guards pass
         const { data: invite, error: claimErr } = await admin
           .from('user_invitations')
           .update({ accepted_at: new Date().toISOString() })
           .eq('token', token)
           .is('accepted_at', null)
           .is('cancelled_at', null)
           .gt('expires_at', new Date().toISOString())
           .select('id, tenant_id, email, role, tenants(id, slug, name)')
           .single()
         if (claimErr || !invite) {
           return { error: 'Este link de convite jĂˇ foi utilizado ou expirou. FaĂ§a login se jĂˇ tiver uma conta.' }
         }

         // Find the auth user that inviteUserByEmail created
         const { data: listRes } = await admin.auth.admin.listUsers()
         const authUser = listRes?.users?.find((u) => u.email?.toLowerCase() === invite.email.toLowerCase())
         if (!authUser) return { error: 'UsuĂˇrio do convite nĂŁo encontrado. PeĂ§a para o administrador reenviar o convite.' }

         const tenantSlug = (invite.tenants as { slug: string } | null)?.slug
         if (!tenantSlug) return { error: 'Corretora nĂŁo encontrada.' }

         // Set password + app_metadata + profile in sequence; best-effort rollback on profile failure
         const { error: updErr } = await admin.auth.admin.updateUserById(authUser.id, {
           password,
           email_confirm: true,
           app_metadata: { tenant_id: invite.tenant_id, role: invite.role, slug: tenantSlug },
           user_metadata: { full_name: fullName },
         })
         if (updErr) return { error: 'Erro ao ativar conta. Tente novamente.' }

         const { error: profErr } = await admin.from('profiles').upsert({
           id: authUser.id,
           tenant_id: invite.tenant_id,
           full_name: fullName,
           role: invite.role,
           active: true,
         })
         if (profErr) return { error: 'Erro ao criar perfil. Tente novamente.' }

         // Auto-login
         const supabase = await createClient()
         const { error: signInErr } = await supabase.auth.signInWithPassword({ email: invite.email, password })
         if (signInErr) redirect(`/login?next=/${tenantSlug}/dashboard`)
         redirect(`/${tenantSlug}/dashboard`)
       }

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Client form â†’ Server Action | Zod schema revalidated server-side; client-only validation cannot be trusted (user can POST directly) |
| BrasilAPI â†’ /api/cnpj proxy | Third-party response is untrusted; never echoed raw into DOM; validated & trimmed to 3 expected fields |
| Invite token â†’ server lookup | Token is bearer credential; DB lookup is the trust check; application-layer role check is secondary defense |
| Admin UI â†’ invite/resend/cancel | `requireAdmin()` gate + RLS `invitations_admin_manage` policy both enforce role='admin' |
| Email service (Supabase auth.admin.inviteUserByEmail) | Relies on Supabase + Resend SMTP reputation; hostile users cannot enumerate emails because inviteUser only works for admins of their own tenant |
| Auth layout â†’ app layout | (auth) routes never render authenticated data; (app) routes require middleware-validated session |
