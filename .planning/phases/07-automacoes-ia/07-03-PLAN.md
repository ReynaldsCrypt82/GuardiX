       ```typescript
       import { redirect } from 'next/navigation'
       import { createClient } from '@/lib/supabase/server'
       import { WebhookConfigForm } from './webhook-config-form'
       import { EmailTemplateForm } from './email-template-form'
       import { WebhookLogsTable } from './webhook-logs-table'
       import { EVENT_TYPE_VALUES } from '@/lib/validations/automation-schemas'

       const EVENT_LABELS: Record<string, string> = {
         policy_expiring: 'Vencimento de apolice',
         financial_overdue: 'Inadimplencia financeira',
         consortium_contemplated: 'Contemplacao de consorcio',
       }

       export default async function AutomacoesPage({ params }: { params: Promise<{ slug: string }> }) {
         const { slug } = await params
         // eslint-disable-next-line @typescript-eslint/no-explicit-any
         const supabase = (await createClient()) as any
         const { data: { user } } = await supabase.auth.getUser()
         const role = (user?.app_metadata as { role?: string })?.role
         if (!user || role !== 'admin') redirect(`/${slug}/dashboard`)

         const [{ data: configs }, { data: templates }, { data: logs }] = await Promise.all([
           supabase.from('webhook_configs').select('id, event_type, url, days_before, active').is('deleted_at', null).order('event_type'),
           supabase.from('email_templates').select('id, event_type, subject, body_html, active').is('deleted_at', null).order('event_type'),
           supabase.from('webhook_logs').select('id, event_type, url_destino, http_status, error_message, triggered_at').order('triggered_at', { ascending: false }).limit(20),
         ])

         const configsByEvent = new Map((configs ?? []).map((c: { event_type: string }) => [c.event_type, c]))
         const templatesByEvent = new Map((templates ?? []).map((t: { event_type: string }) => [t.event_type, t]))

         return (
           <div className="flex flex-col gap-8 p-6">
             <header><h1 className="text-2xl font-semibold">Automacoes</h1><p className="text-sm text-muted-foreground">Configure webhooks n8n, templates de email e visualize logs de disparo.</p></header>

             <section>
               <h2 className="text-lg font-semibold mb-3">Webhooks n8n</h2>
               <div className="grid gap-4 md:grid-cols-3">
                 {EVENT_TYPE_VALUES.map((evt) => (
                   <WebhookConfigForm key={evt} slug={slug} eventType={evt} eventLabel={EVENT_LABELS[evt]} initial={configsByEvent.get(evt)} />
                 ))}
               </div>
             </section>

             <section>
               <h2 className="text-lg font-semibold mb-3">Templates de email</h2>
               <div className="flex flex-col gap-4">
                 {EVENT_TYPE_VALUES.map((evt) => (
                   <EmailTemplateForm key={evt} slug={slug} eventType={evt} eventLabel={EVENT_LABELS[evt]} initial={templatesByEvent.get(evt)} />
                 ))}
               </div>
             </section>

             <section>
               <h2 className="text-lg font-semibold mb-3">Historico de disparos (20 mais recentes)</h2>
               <WebhookLogsTable logs={logs ?? []} eventLabels={EVENT_LABELS} />
             </section>
           </div>
         )
       }
       ```

       import { Table, TableHeader, TableBody, TableRow, TableHead, TableCell } from '@/components/ui/table'
       import { format } from 'date-fns'

       type Log = { id: string; event_type: string; url_destino: string; http_status: number | null; error_message: string | null; triggered_at: string }

                   <TableCell>{eventLabels[l.event_type] ?? l.event_type}</TableCell>
                   <TableCell className="max-w-xs truncate">{l.url_destino}</TableCell>

    5. Criar `src/app/(app)/[slug]/configuracoes/automacoes/test-webhook-button.tsx`:
       ```typescript
       'use client'
       import { useState } from 'react'
       import { Button } from '@/components/ui/button'

       export function TestWebhookButton({ slug, url, eventType }: { slug: string; url: string; eventType: string }) {
         const [result, setResult] = useState<string | null>(null)
         const [loading, setLoading] = useState(false)
         async function handleTest() {
           if (!url) { setResult('Preencha a URL antes de testar.'); return }
           setLoading(true)
           try {
             const res = await fetch(`/api/${slug}/webhook-test`, { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ url, event_type: eventType }) })
             const data = await res.json()
             if (data.error) setResult(`Erro: ${data.error}`)
             else setResult(`HTTP ${data.http_status} ${data.ok ? '(sucesso)' : '(falha)'}`)
           } catch (e) {
             setResult(`Erro de rede: ${String(e)}`)
           } finally {
             setLoading(false)
           }
         }
         return (
           <div className="flex flex-col gap-1">
             <Button type="button" variant="outline" size="sm" onClick={handleTest} disabled={loading}>{loading ? 'Testando...' : 'Testar webhook'}</Button>
             {result && <p className="text-xs text-muted-foreground">{result}</p>}
           </div>
         )
       }
       ```
